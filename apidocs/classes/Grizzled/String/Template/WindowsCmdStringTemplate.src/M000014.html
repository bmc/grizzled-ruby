            def handle_match(m, s)
              if (m[1].length % 2) == 1
                # Odd number of backslashes before "%". Valid escape.

                b = m.begin(0)
                start = (b == 0 ? "" : s[0..(b-1)])
                start + ESCAPED_PERCENT_PLACEHOLDER + pre_sub(s[m.end(0)..-1])
              else
                # Even number of backslashes before "%". Not an escape.
                s
              end
            end
