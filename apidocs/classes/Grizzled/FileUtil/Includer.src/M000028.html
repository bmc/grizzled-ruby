      def each_line(ignored=nil)
        if not @closed
          if @temp.nil?
            preprocess(@source)
          end

          File.open(@temp.path) do |f|
            f.each_line do |line|
              yield line
            end
          end

          @temp.close
          @temp.unlink
          @temp = nil
          @closed = true
        end
      end
