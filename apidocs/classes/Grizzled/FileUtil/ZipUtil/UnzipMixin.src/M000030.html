        def zip_file_entries(zip_file, &block)
          if not block_given?
            a = []
          end

          Zip::ZipFile.foreach(zip_file) do |entry|
            if block_given?
              block.call(entry.to_s)
            else
              a << entry.to_s
            end
          end

          if not block_given?
            return a.each
          end
        end
